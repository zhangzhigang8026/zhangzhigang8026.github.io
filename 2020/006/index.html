栈（Go实现）

概念

栈是一种典型的先进后出( First in Last Out )的数据结构，其操作主要有压栈(push)与出栈(pop)两种操作，如下图所示（维基百科）。两种操作都操作栈顶，当然，它也有栈底。

实现

下面主要介绍了3种实现：

  • 顺序栈
  • 双向顺序栈
  • 链栈

顺序栈

package SqStack

import (
	"errors"
	"fmt"
)

type SqStack struct {
	Data []interface{}
	Top  int
	Size int
}

func NewSqStack(size int) *SqStack {
	return &SqStack{
		Data: nil,
		Top:  -1,
		Size: size,
	}
}

func (s *SqStack) Display() {
	if s.Top == -1 {
		fmt.Println("空栈")
		return
	}

	fmt.Println("栈顶指针：", s.Top)
	fmt.Println("栈数据：", s.Data[:s.Top+1])
}

// Push 压栈操作
func (s *SqStack) Push(e interface{}) error {
	// 空值
	if len(s.Data) == 0 {
		s.Data = make([]interface{}, s.Size)
		s.Data[0] = e
	}

	// 正常情况
	if s.Top == s.Size-1 {
		fmt.Println("栈已满")
		return errors.New("栈已满")
	}

	s.Data[s.Top+1] = e
	fmt.Println(s.Data)
	s.Top++
	return nil
}

func (s *SqStack) Pop() (interface{}, error) {
	if s.Top == -1 {
		fmt.Println("空栈无法pop")
		return nil, errors.New("空栈无法pop")
	}

	e := s.Data[s.Top]
	s.Top--
	return e, nil
}

双向顺序栈

package ShStack

import (
	"errors"
	"fmt"
)

type ShStack struct {
	Data			[]interface{}			// 数组切片
	Top1			int					// 栈1栈顶指针
	Top2			int					// 栈2栈顶指针
	Size			int					// 栈的最大容量
}

func NewShStack(size int) *ShStack {
	return &ShStack{
		Data: nil,
		Top1: -1,
		Top2: size,
		Size: size,
	}
}

func (s *ShStack) Display() {
	if s.Top1 == -1 {
		fmt.Println("栈1为空栈")
	} else {
		fmt.Println("栈1栈顶指针：", s.Top1)
		fmt.Println("栈1数据：", s.Data[:s.Top1+1])
	}

	if s.Top2 == s.Size {
		fmt.Println("栈2为空栈")
	} else {
		fmt.Println("栈2栈顶指针：", s.Top2)
		fmt.Println("栈2数据：", s.Data[s.Top2:])
	}
}

// 进栈操作
// num 表示元素 e Push到栈1还是栈2
func (s *ShStack) Push(e interface{}, num int) error {
	if num < 1 || num > 2 {
		fmt.Println("栈选择错误")
		return errors.New("栈选择错误")
	}

	if len(s.Data) == 0 {
		s.Data = make([]interface{}, s.Size)
		s.Data[0] = e
	}

	if s.Top1 + 1 == s.Top2 {
		fmt.Println("空间已满")
		return errors.New("空间已满")
	}

	// 对第一个栈进行入栈
	if num == 1 {
		s.Data[s.Top1 + 1] = e
		s.Top1 ++
	} else {	// 对第二个栈进行入栈
		s.Data[s.Top2 - 1] = e
		s.Top2 --
	}
	return nil
}

// 出栈操作
func (s *ShStack)Pop(num int) (interface{}, error) {
	if (num == 1 && s.Top1 == -1) || (num ==2 && s.Top2 == s.Size) {
		fmt.Println("空栈无法pop")
		return nil, errors.New("空栈无法pop")
	}

	var e interface{}

	if num == 1 {
		e = s.Data[s.Top1]
		s.Top1 --
	} else {
		e = s.Data[s.Top2]
		s.Top2 --
	}

	return e, nil

}

链栈

package LinkStack

import (
	"errors"
	"fmt"
)

type Node struct {
	Data			interface{}
	Next			*Node
}

type LinkStack struct {
	Top				*Node		// 栈顶指针
	Length			int
}

func NewLinkStack() *LinkStack {
	return &LinkStack{
		Top:    nil,
		Length: 0,
	}
}

func (l *LinkStack) Display()  {
	if l.Top == nil {
		fmt.Println("空栈")
		return
	}

	fmt.Println("栈顶指针数据：", l.Top.Data)
	fmt.Printf("栈数据：")
	currentNode := l.Top
	for i := 1; i <= l.Length; i++ {
		fmt.Printf("%v ", currentNode.Data)
		currentNode = currentNode.Next
	}
	fmt.Println()
}

func (l *LinkStack) Push(e interface{})  {
	node := &Node{
		Data: e,
		Next: l.Top,
	}
	l.Top = node
	l.Length ++
}

func (l *LinkStack) Pop() (e interface{}, err error) {
	if l.Top == nil {
		fmt.Println("空栈无法pop")
		return nil, errors.New("空栈无法pop")
	}

	e = l.Top.Data
	node := l.Top
	l.Top = node.Next
	l.Length --
	return e, nil
}
